#include<iostream>
#include<string>
using namespace std;
int plus(int x, int y);
int minus(int x, int y);
int divide(int x, int y);
int multi(int x, int y);
int main()
{
	int x=100,y=20;
	int result;
	result=plus(x,y);
	cout<<"Plus = "<<result<<endl;
	result=minus(x,y);
	cout<<"Minus = "<<result<<endl;
	result=divide(x,y);
	cout<<"Divide = "<<result<<endl;
	result=multi(x,y);
	cout<<"Multi = "<<result<<endl;
	return(0);
}
int plus(int x, int y)
{
	int sum=x+y;
	return sum;
}
int minus(int x, int y)
{
	int sum=x-y;
	return sum;
}
int divide(int x, int y)
{
	int sum=x/y;
	return sum;
}
int multi(int x,int y) //x, y to receieve line 8 in same position
{
	int sum=x*y;
	return sum; //return value to calling position
}
